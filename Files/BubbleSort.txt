public class BubbleSort {

	public static void bsort(Comparable[] a){
		boolean sw=true;
		Comparable tmp;
		int j=0;
		while(sw){
			sw=false;
			j++;
			for(int i=0;i<a.length-j;i++){
				if(a[i].compareTo(a[i+1])>0){
					tmp=a[i];
					a[i]=a[i+1];
					a[i+1]=tmp;
					sw=true;
				}
				
			}
			
		}
		
	}
	
}
